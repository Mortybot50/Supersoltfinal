C) API: Create POs from Order Guide (grouped by supplier)

export const runtime = "nodejs";
import { NextResponse } from "next/server";
import { requireOrg } from "@/lib/authz";
import { db } from "@/db";
// import your order guide calculator: lib/inventory-ordering
// import ingredient_suppliers, ingredients, purchase_orders, purchase_order_lines

// Input: ?start=YYYY-MM-DD&end=YYYY-MM-DD  (or start+days; use same window code you already use)
export async function POST(req: Request) {
  const { venueId, userId } = await requireOrg({ role: "manager" });
  const url = new URL(req.url);
  const start = url.searchParams.get("start");
  const end = url.searchParams.get("end");
  if (!start || !end) return NextResponse.json({ error: "missing_window" }, { status: 400 });

  // 1) Build order guide items with your existing logic (shortfalls grouped by supplier)
  // Expected: [{ supplierName, supplierEmail?, items: [{ ingredientId, recommendedPacks, packSize, packUnit, packCostCents }] }]
  const guides = await buildOrderGuideForWindow({ venueId, startISO: start, endISO: end }); // <-- use your lib

  const created:any[] = [];

  for (const g of guides) {
    if (!g.items || g.items.length === 0) continue;

    // 2) Create PO (draft)
    const [po] = await db.insert(purchase_orders).values({
      venue_id: venueId,
      supplier_name: g.supplierName,
      supplier_email: g.supplierEmail ?? null,
      status: "draft",
      order_date: new Date(start),
      expected_date: null,
      notes: `Auto-created from Order Guide (${start}..${end})`,
      subtotal_cents: 0, tax_cents: 0, total_cents: 0,
      created_by: userId
    }).returning();

    let subtotal = 0;

    for (const it of g.items) {
      const packQty = Math.max(0, Math.ceil(it.recommendedPacks ?? 0));
      if (packQty === 0) continue;

      // unit snapshot from pack
      const packSizeBase =
        it.packUnit === "kg" ? it.packSize * 1000 :
        it.packUnit === "l"  ? it.packSize * 1000 :
        it.packUnit === "g" || it.packUnit === "ml" ? it.packSize :
        it.packUnit === "each" ? it.packSize : it.packSize;

      const unitCostSnap = Math.round(it.packCostCents / Math.max(1, packSizeBase));
      const extended = packQty * it.packCostCents;

      await db.insert(purchase_order_lines).values({
        po_id: po.id,
        ingredient_id: it.ingredientId,
        supplier_id: it.supplierId ?? null,
        pack_qty: packQty,
        pack_size: it.packSize,
        pack_unit: it.packUnit,
        pack_cost_cents: it.packCostCents,
        unit_cost_cents_snapshot: unitCostSnap,
        extended_cents: extended,
        note: it.note ?? null
      });

      subtotal += extended;
    }

    await db.update(purchase_orders).set({
      subtotal_cents: subtotal,
      tax_cents: 0,
      total_cents: subtotal
    }).where(purchase_orders.id.eq(po.id));

    created.push(po);
  }

  return NextResponse.json({ ok:true, purchaseOrders: created }, { status: 201 });
}

import { purchase_orders } from "@/db/schema";

export async function GET(req: Request) {
  const { venueId } = await requireOrg({ role: "supervisor" });
  const url = new URL(req.url);
  const status = url.searchParams.get("status"); // optional

  const rows = await db.query.purchase_orders.findMany({
    where: (t,{eq,and}) => status
      ? and(eq(t.venue_id, venueId), eq(t.status, status as any))
      : eq(t.venue_id, venueId),
    orderBy: (t,{desc}) => [desc(t.order_date), desc(t.created_at)],
    limit: 200
  });

  return NextResponse.json({ rows });
}

export const runtime = "nodejs";
import { NextResponse } from "next/server";
import { requireOrg } from "@/lib/authz";
import { db } from "@/db";
import { purchase_orders, purchase_order_lines, ingredients } from "@/db/schema";
import { eq, and } from "drizzle-orm";

export async function GET(_:Request, { params }:{params:{id:string}}) {
  const { venueId } = await requireOrg();
  const poId = params.id;

  const po = await db.query.purchase_orders.findFirst({
    where: (t,{and,eq}) => and(eq(t.venue_id, venueId), eq(t.id, poId))
  });
  if (!po) return NextResponse.json({ error:"not_found" }, { status:404 });

  const lines = await db.select({
    id: purchase_order_lines.id,
    ingredientId: purchase_order_lines.ingredient_id,
    ingredientName: ingredients.name,
    packQty: purchase_order_lines.pack_qty,
    packSize: purchase_order_lines.pack_size,
    packUnit: purchase_order_lines.pack_unit,
    packCostCents: purchase_order_lines.pack_cost_cents,
    extendedCents: purchase_order_lines.extended_cents
  })
  .from(purchase_order_lines)
  .leftJoin(ingredients, eq(ingredients.id, purchase_order_lines.ingredient_id))
  .where(eq(purchase_order_lines.po_id, poId));

  return NextResponse.json({ po, lines });
}

	4.	Receive goods (partial/complete; updates stock & last cost)
File: app/api/purchases/[id]/receive/route.ts (POST)

export const runtime = "nodejs";
import { NextResponse } from "next/server";
import { requireOrg } from "@/lib/authz";
import { db } from "@/db";
import { purchase_orders, purchase_order_lines, receipts, receipt_lines, stock_movements, ingredients } from "@/db/schema";
import { eq, and } from "drizzle-orm";

type LineIn = { poLineId:string; packsReceived:number };

export async function POST(req:Request, { params }:{params:{id:string}}) {
  const { venueId, userId } = await requireOrg({ role:"supervisor" });
  const body = await req.json() as { lines: LineIn[]; note?:string };

  const poId = params.id;
  const po = await db.query.purchase_orders.findFirst({
    where: (t,{and,eq}) => and(eq(t.venue_id, venueId), eq(t.id, poId))
  });
  if (!po) return NextResponse.json({ error:"not_found" }, { status:404 });

  const allLines = await db.query.purchase_order_lines.findMany({ where: (t,{eq}) => eq(t.po_id, poId) });

  const [rcpt] = await db.insert(receipts).values({
    po_id: poId, received_by: userId, status: "partial", note: body.note ?? null
  }).returning();

  let fullyReceived = true;

  for (const l of body.lines) {
    const line = allLines.find(x => String(x.id) === String(l.poLineId));
    if (!line) continue;

    // Convert packs to base units
    const sizeBase =
      line.pack_unit === "kg" ? Number(line.pack_size) * 1000 :
      line.pack_unit === "l"  ? Number(line.pack_size) * 1000 :
      (line.pack_unit === "g" || line.pack_unit === "ml") ? Number(line.pack_size) :
      line.pack_unit === "each" ? Number(line.pack_size) : Number(line.pack_size);

    const qtyBase = Number(l.packsReceived) * sizeBase;

    // Stock movement
    await db.insert(stock_movements).values({
      venue_id: venueId,
      ingredient_id: line.ingredient_id,
      qty_base_units: qtyBase,
      unit: line.pack_unit,
      movement_type: "receiving",
      ref_id: rcpt.id
    });

    // Snapshot cost per base unit at receive time (from line snapshot if available)
    const unitCost = line.unit_cost_cents_snapshot ?? Math.round(line.pack_cost_cents / Math.max(1, sizeBase));

    await db.insert(receipt_lines).values({
      receipt_id: rcpt.id,
      po_line_id: line.id,
      ingredient_id: line.ingredient_id,
      packs_received: l.packsReceived,
      qty_base_units: qtyBase,
      unit_cost_cents_snapshot: unitCost
    });

    // Update ingredient last cost (simple MVP rule: set to last received unit cost)
    await db.update(ingredients).set({ cost_per_unit_cents: unitCost }).where(and(eq(ingredients.id, line.ingredient_id), eq(ingredients.venue_id, venueId)));

    // Track if still outstanding (simple rule: if received packs < ordered packs → not full)
    if (Number(l.packsReceived) < Number(line.pack_qty)) fullyReceived = false;
  }

  await db.update(receipts).set({ status: fullyReceived ? "full" : "partial" }).where(eq(receipts.id, rcpt.id));
  await db.update(purchase_orders).set({ status: fullyReceived ? "received" : "receiving" }).where(eq(purchase_orders.id, poId));

  return NextResponse.json({ ok:true, receiptId: rcpt.id, status: fullyReceived ? "received" : "receiving" });
}
