2) Wire a Generate endpoint that uses DateNav window

Edit: app/api/ops/suggestions/generate/route.ts

export const runtime = "nodejs";
import { NextResponse } from "next/server";
import { requireOrg } from "@/lib/authz";
import { getWindow } from "@/lib/date-window";
import { runGuardrails } from "@/lib/guardrails"; // make sure this exists/exports

export async function POST(req: Request) {
  const { venueId } = await requireOrg({ role: "manager" });
  const url = new URL(req.url);
  const period = (url.searchParams.get("period") || "week") as "day"|"week"|"month";
  const startISO = url.searchParams.get("start") || new Date().toISOString();
  const { start, end } = getWindow(period, startISO);

  const suggestions = await runGuardrails({ start, end, period, venueId });
  // persist NEW suggestions into ops_suggestions with payload JSONB if not already present (idempotent on (venueId,type,title,start,end))
  // return list of NEW suggestions
  return NextResponse.json({ suggestions });
}


Ensure idempotency: don’t create duplicates for the same (venue, type, title, window).

3) Add a Generate button + empty state in the panel

Edit: components/SuggestionsPanel.tsx (or wherever you render suggestions)

Read period/start from the same state your DateNav controls.

Add a button:

"use client";
import { useState } from "react";
import { useRouter, useSearchParams } from "next/navigation";

export function SuggestionsPanel() {
  const [loading, setLoading] = useState(false);
  const router = useRouter();
  const sp = useSearchParams();
  const period = sp.get("period") ?? "week";
  const start = sp.get("start") ?? new Date().toISOString();

  async function generate() {
    setLoading(true);
    const qs = new URLSearchParams({ period, start });
    const res = await fetch(`/api/ops/suggestions/generate?`+qs.toString(), { method: "POST" });
    setLoading(false);
    if (res.ok) router.refresh(); // re-fetch list
  }

  return (
    <div className="space-y-2">
      <div className="flex items-center justify-between">
        <h3 className="font-medium">Suggestions</h3>
        <button onClick={generate} disabled={loading} className="rounded-md border px-3 py-1.5 text-sm">
          {loading ? "Generating…" : "Generate"}
        </button>
      </div>
      {/* existing list rendering */}
      {/* empty state */}
      {/* if list length === 0, show a hint: */}
      <div className="text-sm text-muted-foreground">
        No suggestions yet. Add costs/recipes/schedule, then click <b>Generate</b>.
      </div>
    </div>
  );
}
