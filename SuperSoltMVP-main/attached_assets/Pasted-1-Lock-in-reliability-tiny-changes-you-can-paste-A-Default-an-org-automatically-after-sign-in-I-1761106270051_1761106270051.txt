useEffect(() => {

C. Add “Cancel invite” (soft delete)
API:

// app/api/people/invites/[id]/route.ts
import { NextRequest, NextResponse } from "next/server"
import { db } from "@/db"
import { invites } from "@/db/schema"
import { eq, and } from "drizzle-orm"
import { requireRole } from "@/lib/authz"
import { cookies } from "next/headers"

export async function DELETE(
  _req: NextRequest,
  { params }: { params: { id: string } }
) {
  const cookieStore = await cookies()
  const orgId = cookieStore.get("orgId")?.value
  if (!orgId) return NextResponse.json({ error: "No org" }, { status: 400 })
  await requireRole(orgId, ["owner","manager"])
  await db.update(invites).set({ status: "cancelled" }).where(and(eq(invites.id, params.id), eq(invites.orgId, orgId)))
  return NextResponse.json({ ok: true })
}

2) Make Playwright happy with shadcn/Radix Select

Playwright often struggles with Select portals. Use this helper:

// tests/helpers/select.ts
export async function selectRadixOption(page, triggerText: string, optionText: string) {
  // open the select (by label or button text)
  await page.getByRole("button", { name: triggerText }).click()
  // Radix puts content in a portal; target role=option or the item within viewport
  const option = page.locator('[data-radix-select-viewport] [data-highlighted], [role="option"]', { hasText: optionText })
  await option.first().click()
}


Example in your test:

await selectRadixOption(page, "Select organization", "Demo Bistro")
await page.getByRole("button", { name: "Create Invite" }).click()
await expect(page.getByText("Invite created")).toBeVisible()


If needed, force the portal root visible to Playwright:

test.beforeEach(async ({ page }) => {
  await page.addStyleTag({ content: `*, *::before, *::after { caret-color: transparent !important; }` })
})
