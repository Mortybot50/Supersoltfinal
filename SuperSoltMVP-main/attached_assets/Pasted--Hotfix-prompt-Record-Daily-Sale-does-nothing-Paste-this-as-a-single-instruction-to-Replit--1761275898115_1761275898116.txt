   export function RecordSaleForm({ defaultDate, items }: { defaultDate: string; items: { id:string; name:string; priceCents:number }[] }) {

     const [date, setDate] = useState(defaultDate);

         console.error(msg);
         alert("Could not record sale");
       }
     }

     return (
       <form onSubmit={onSubmit} className="space-y-3">
         {/* wire your date picker, select, input to setDate/setMenuItemId/setQty */}

2) API route: create/verify POST /api/sales/daily (Node runtime).
   - File: app/api/sales/daily/route.ts
   - Must:
     a) `export const runtime = "nodejs";`
     b) `await requireOrg()` to get orgId/venueId and userId,
     c) accept JSON { date, menuItemId, quantity },
     d) upsert/insert into your daily sales table (by venueId + date + menuItemId),
     e) return 200 JSON,
     f) kick an async forecast generation for that date (idempotent).

   Example skeleton (adapt table names/ORM):
   -------------------------------------------------
   export const runtime = "nodejs";
   import { NextResponse } from "next/server";
   import { requireOrg } from "@/lib/authz";
   import { db } from "@/db";
   import { daily_sales } from "@/db/schema"; // adjust
   import { eq, and } from "drizzle-orm";

   export async function POST(req: Request) {
     const { venueId } = await requireOrg({ role: "manager" });
     const { date, menuItemId, quantity } = await req.json();
     if (!venueId || !date || !menuItemId || !quantity) {
       return NextResponse.json({ error: "bad_request" }, { status: 400 });
     }
     const day = new Date(date); day.setHours(0,0,0,0);

     // Upsert: if a row for (venueId, date, menuItemId) exists, add to quantity; else insert
     const existing = await db.query.daily_sales.findFirst({
       where: (t, { and, eq }) => and(eq(t.venue_id, venueId), eq(t.date, day), eq(t.menu_item_id, menuItemId))
     });

     if (existing) {
       await db.update(daily_sales)
         .set({ quantity: existing.quantity + Number(quantity) })
         .where(and(eq(daily_sales.venue_id, venueId), eq(daily_sales.date, day), eq(daily_sales.menu_item_id, menuItemId)));
     } else {
       await db.insert(daily_sales).values({
         venue_id: venueId, date: day, menu_item_id: menuItemId, quantity: Number(quantity)
       });
     }

     // Fire-and-forget: ensure forecasts exist for that day window
     const qs = new URLSearchParams({ start: day.toISOString(), days: "1" });
     fetch(`${process.env.NEXT_PUBLIC_APP_URL ?? ""}/api/forecast/generate?` + qs, { method: "POST", cache: "no-store" }).catch(()=>{});

     return NextResponse.json({ ok: true });
   }
   -------------------------------------------------
