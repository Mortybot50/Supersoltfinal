a) POST app/api/waste/route.ts

export const runtime = "nodejs";
import { NextResponse } from "next/server";
import { requireOrg } from "@/lib/authz";
import { toBase } from "@/lib/units";
import { getIngredientUnitCostCentsSnapshot } from "@/lib/costing";
import { db } from "@/db";
import { waste_events, stock_movements } from "@/db/schema"; // adjust paths
import { z } from "zod";

const Body = z.object({
  date: z.string(), // YYYY-MM-DD
  ingredientId: z.string(),
  qty: z.coerce.number().positive(),
  unit: z.enum(["g","kg","ml","l","each"]),
  reason: z.enum(["prep","spoilage","overportion","transfer","theft","other"]).default("spoilage"),
  note: z.string().optional()
});

export async function POST(req: Request) {
  try {
    const { userId, venueId } = await requireOrg({ role: "supervisor" }); // allow manager/supervisor
    const body = Body.parse(await req.json());
    const day = new Date(body.date); day.setHours(0,0,0,0);

    const base = toBase(body.qty, body.unit);
    const unitCostCents = await getIngredientUnitCostCentsSnapshot({ venueId, ingredientId: body.ingredientId, at: day });

    // insert event
    const [evt] = await db.insert(waste_events).values({
      venue_id: venueId,
      date: day,
      ingredient_id: body.ingredientId,
      qty: body.qty,
      unit: body.unit,
      reason: body.reason,
      note: body.note ?? null,
      unit_cost_cents_snapshot: unitCostCents ?? null,
      created_by: userId
    }).returning();

    // stock movement (negative)
    await db.insert(stock_movements).values({
      venue_id: venueId,
      ingredient_id: body.ingredientId,
      qty_base_units: -base.qty,
      unit: body.unit,
      movement_type: "waste",
      ref_id: evt.id
    });

    return NextResponse.json({ ok:true, event: evt, costCents: unitCostCents ? Math.round(unitCostCents * base.qty) : null }, { status: 201 });
  } catch (e:any) {
    return NextResponse.json({ error: e?.message ?? "waste_create_failed" }, { status: e?.statusCode ?? 500 });
  }
}

b) GET app/api/waste/route.ts

export const runtime = "nodejs";
import { NextResponse } from "next/server";
import { requireOrg } from "@/lib/authz";
import { db } from "@/db";
import { waste_events, ingredients } from "@/db/schema";
import { and, gte, lte, eq } from "drizzle-orm";
import { z } from "zod";

export async function GET(req: Request) {
  try {
    const { venueId } = await requireOrg();
    const url = new URL(req.url);
    const start = url.searchParams.get("start");
    const end = url.searchParams.get("end");
    const StartEnd = z.object({ start:z.string(), end:z.string() });
    const { start: s, end: e } = StartEnd.parse({ start, end });

    const startD = new Date(s); startD.setHours(0,0,0,0);
    const endD = new Date(e);   endD.setHours(23,59,59,999);

    // join ingredient name for display
    const rows = await db.select({
      id: waste_events.id,
      date: waste_events.date,
      ingredientId: waste_events.ingredient_id,
      ingredientName: ingredients.name,
      qty: waste_events.qty,
      unit: waste_events.unit,
      reason: waste_events.reason,
      unitCostCents: waste_events.unit_cost_cents_snapshot,
      note: waste_events.note,
    })
    .from(waste_events)
    .leftJoin(ingredients, eq(ingredients.id, waste_events.ingredient_id))
    .where(and(eq(waste_events.venue_id, venueId), gte(waste_events.date, startD), lte(waste_events.date, endD)))
    .orderBy(waste_events.date);

    // aggregates
    const totalCostCents = rows.reduce((s, r) => s + (r.unitCostCents ? Math.round(r.unitCostCents * (r.unit==="kg"?r.qty*1000: r.unit==="l"?r.qty*1000: r.unit==="g"||r.unit==="ml"?r.qty: r.qty)) : 0), 0);
    const byIngredient = Object.values(rows.reduce((acc:any, r) => {
      const key = r.ingredientId || "unknown";
      const add = r.unitCostCents ? Math.round(r.unitCostCents * (r.unit==="kg"?r.qty*1000: r.unit==="l"?r.qty*1000: r.unit==="g"||r.unit==="ml"?r.qty: r.qty)) : 0;
      acc[key] ??= { ingredientId:key, ingredientName:r.ingredientName ?? "Unknown", costCents:0 };
      acc[key].costCents += add;
      return acc;
    }, {}));

    return NextResponse.json({ items: rows, totals: { totalCostCents, byIngredient } }, { status: 200 });
  } catch (e:any) {
    return NextResponse.json({ items: [], totals: { totalCostCents:0, byIngredient: [] } }, { status: 200 });
  }
}

c) DELETE app/api/waste/[id]/route.ts

export const runtime = "nodejs";
import { NextResponse } from "next/server";
import { requireOrg } from "@/lib/authz";
import { db } from "@/db";
import { waste_events, stock_movements } from "@/db/schema";
import { eq, and } from "drizzle-orm";

export async function DELETE(_: Request, { params }: { params: { id: string }}) {
  try {
    const { venueId } = await requireOrg({ role:"manager" });
    const id = params.id;
    // restore stock by deleting movement and inserting reverse
    const [evt] = await db.delete(waste_events).where(and(eq(waste_events.id, id), eq(waste_events.venue_id, venueId))).returning();
    if (evt) {
      await db.delete(stock_movements).where(and(eq(stock_movements.ref_id, id), eq(stock_movements.venue_id, venueId)));
      // optional: insert a compensating movement type "waste_revert" with +qty_base_units
    }
    return NextResponse.json({ ok:true });
  } catch (e:any) {
    return NextResponse.json({ error: e?.message ?? "waste_delete_failed" }, { status: e?.statusCode ?? 500 });
  }
}
