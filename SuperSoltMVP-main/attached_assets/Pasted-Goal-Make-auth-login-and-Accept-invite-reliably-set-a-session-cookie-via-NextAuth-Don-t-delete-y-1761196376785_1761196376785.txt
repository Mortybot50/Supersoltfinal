import NextAuth from "next-auth";
import Credentials from "next-auth/providers/credentials";
import { db } from "@/db";
import { users, memberships } from "@/db/schema";
import { eq } from "drizzle-orm";
import bcrypt from "bcrypt";

const handler = NextAuth({
  session: { strategy: "jwt" },
  providers: [
    Credentials({
      name: "Email & Password",
      credentials: { email: {}, password: {} },
      async authorize(creds) {
        const email = String(creds?.email || "").toLowerCase();
        const password = String(creds?.password || "");
        const user = await db.query.users.findFirst({ where: (t, { eq }) => eq(t.email, email) });
        if (!user || !user.password_hash) return null;
        const ok = await bcrypt.compare(password, user.password_hash);
        if (!ok) return null;
        // pick first membership as default org
        const m = await db.query.memberships.findFirst({ where: (t,{eq}) => eq(t.user_id, user.id) });
        if (!m) return null;
        return { id: user.id, email: user.email, name: user.name, orgId: m.org_id };
      }
    })
  ],
  callbacks: {
    async jwt({ token, user }) {
      if (user) {
        token.orgId = (user as any).orgId;
      }
      return token;
    },
    async session({ session, token }) {
      (session as any).orgId = (token as any).orgId;
      return session;
    }
  }
});

export { handler as GET, handler as POST };

2) Update Login page to call NextAuth directly

File: app/auth/login/page.tsx

"use client";
import { useState } from "react";
import { signIn } from "next-auth/react";
import { useSearchParams } from "next/navigation";

export default function LoginPage() {
  const [msg, setMsg] = useState<string | null>(null);
  const params = useSearchParams();
  const prefill = params.get("email") || "";

  async function submit(form: FormData) {
    const email = String(form.get("email"));
    const password = String(form.get("password"));
    const res = await signIn("credentials", { redirect: false, email, password, callbackUrl: "/dashboard" });
    if (res?.ok) window.location.href = "/dashboard";
    else setMsg("Invalid credentials.");
  }

  return (
    <div className="mx-auto max-w-md p-6">
      <h1 className="text-xl font-semibold">Sign in</h1>
      <form action={submit} className="mt-4 space-y-3">
        <input name="email" defaultValue={prefill} placeholder="you@venue.com" className="w-full rounded-md border p-2" required />
        <input name="password" type="password" placeholder="Password" className="w-full rounded-md border p-2" required />
        <button className="rounded-md bg-[hsl(var(--accent))] px-3 py-2 text-white">Sign in</button>
      </form>
      {msg && <div className="mt-2 text-sm text-rose-600">{msg}</div>}
      <a href="/auth/forgot" className="mt-2 inline-block text-sm text-muted-foreground hover:underline">Forgot password?</a>
    </div>
  );
}

"use client";
import { useState } from "react";
import { signIn } from "next-auth/react";

export default function AcceptForm({ token }: { token: string }) {
  const [loading, setLoading] = useState(false);
  const [msg, setMsg] = useState<string | null>(null);

  async function submit(form: FormData) {
    setLoading(true);
    const password = String(form.get("password"));
    const name = String(form.get("name"));
    const res = await fetch("/api/auth/accept", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ token, name, password })
    });
    setLoading(false);

    if (!res.ok) { setMsg("Invitation invalid or expired."); return; }

    const { email } = await res.json();
    const s = await signIn("credentials", { redirect: false, email, password, callbackUrl: "/dashboard" });
    if (s?.ok) window.location.href = "/dashboard";
    else window.location.href = `/auth/login?email=${encodeURIComponent(email)}`;
  }

  return (
    <form action={submit} className="mt-4 space-y-3">
      <input name="name" placeholder="Full name" className="w-full rounded-md border p-2" required />
      <input name="password" type="password" placeholder="Create password" className="w-full rounded-md border p-2" required minLength={8}/>
      <button disabled={loading} className="rounded-md bg-[hsl(var(--accent))] px-3 py-2 text-white">{loading?"Working...":"Join"}</button>
      {msg && <div className="text-sm text-rose-600">{msg}</div>}
    </form>
  );
}


This guarantees a proper NextAuth session cookie on both login and invite acceptance.
