export const runtime = "nodejs";
import Papa from "papaparse";
import { z } from "zod";
import { NextResponse } from "next/server";
import { requireOrg } from "@/lib/authz";
import { db } from "@/db";

const Row = z.object({
  date: z.string().transform(s => new Date(s)),
  revenue_cents: z.coerce.number().int().nonnegative(),
  orders: z.coerce.number().int().nonnegative().optional(),
  covers: z.coerce.number().int().nonnegative().optional()
});

export async function POST(req: Request) {
  const { venueId } = await requireOrg({ role: "manager" });
  const text = await req.text(); // CSV in body
  const { data } = Papa.parse<string[]>(text, { header: true, skipEmptyLines: true });
  let inserted = 0, updated = 0; const errors:any[] = [];

  for (let i=0; i<data.length; i++) {
    try {
      const row = Row.parse(data[i]);
      // upsert sales by (venueId, date)
      // await db.insert(...).onConflictDoUpdate(...)
      inserted++; // or updated++
    } catch (e:any) {
      errors.push({ row: i+2, msg: e.message });
    }
  }
  return NextResponse.json({ inserted, updated, skipped: 0, errors });
}
