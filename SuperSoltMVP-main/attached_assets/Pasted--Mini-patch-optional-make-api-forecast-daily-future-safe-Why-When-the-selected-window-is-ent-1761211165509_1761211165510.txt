2) PostHog analytics (client‑only, no PII)

File: lib/analytics.tsx

"use client";
import posthog from "posthog-js";
import { useEffect } from "react";

export function AnalyticsProvider({ user, children }:{ user?: { id:string; orgId?:string; role?:string }, children: React.ReactNode }) {
  useEffect(() => {
    if (!process.env.NEXT_PUBLIC_POSTHOG_KEY) return;
    posthog.init(process.env.NEXT_PUBLIC_POSTHOG_KEY!, { api_host: process.env.NEXT_PUBLIC_POSTHOG_HOST || "https://app.posthog.com" });
    if (user?.id) {
      posthog.identify(user.id, { orgId: user.orgId, role: user.role }); // no email/PII
    }
    return () => { try { posthog.shutdown?.(); } catch(_){} };
  }, [user?.id, user?.orgId, user?.role]);

  return <>{children}</>;
}

// small helpers
export const track = (name: string, props?: Record<string, any>) => {
  try { posthog.capture(name, props); } catch {}
};

4) Minimal internal “Usage” page (no PII)

File: app/internal/usage/page.tsx (manager/owner only; use your existing requireOrg()/role checks)

import { requireOrg } from "@/lib/authz";
import { db } from "@/db";
import { ops_suggestions, shifts, sales_forecasts } from "@/db/schema";
import { sql } from "drizzle-orm";

export default async function UsagePage() {
  await requireOrg({ role: "manager" }); // restrict
  const [sug, ro, fc] = await Promise.all([
    db.execute(sql`SELECT status, COUNT(*) c FROM ops_suggestions GROUP BY status`),
    db.execute(sql`SELECT COUNT(*) c FROM shifts WHERE start_at > now() - interval '7 days'`),
    db.execute(sql`SELECT COUNT(*) c FROM sales_forecasts WHERE date > now() - interval '7 days'`),
  ]);

  return (
    <div className="p-6 space-y-4">
      <h1 className="text-xl font-semibold">Staging usage (last 7 days)</h1>
      <div className="grid grid-cols-1 gap-4 md:grid-cols-3">
        <div className="card p-4"><div className="text-sm">Suggestions</div><div className="text-2xl font-semibold">{sug.rows.reduce((a:any,r:any)=>a+Number(r.c),0)}</div></div>
        <div className="card p-4"><div className="text-sm">Draft/Published Shifts</div><div className="text-2xl font-semibold">{ro.rows[0]?.c ?? 0}</div></div>
        <div className="card p-4"><div className="text-sm">Forecast rows</div><div className="text-2xl font-semibold">{fc.rows[0]?.c ?? 0}</div></div>
      </div>
      <p className="text-xs text-muted-foreground">No PII. For full analytics, see PostHog.</p>
    </div>
  );
}
