import { NextResponse } from "next/server";
import { db } from "@/server/db"; // adjust import to your Drizzle instance
import { sql, and, eq, gte, lte } from "drizzle-orm";
import { daily_sales, menu_items, shifts, staff } from "@/server/schema"; // adjust paths/names as in repo

// Helper: get start of week (Mon) in server TZ
function startOfWeek(d = new Date()) {
  const day = d.getDay(); // 0..6 Sun..Sat
  const diff = (day === 0 ? -6 : 1) - day; // move to Monday
  const s = new Date(d);
  s.setHours(0,0,0,0);
  s.setDate(s.getDate() + diff);
  return s;
}
function startOfDay(d = new Date()) {
  const s = new Date(d); s.setHours(0,0,0,0); return s;
}
function endOfDay(d = new Date()) {
  const e = new Date(d); e.setHours(23,59,59,999); return e;
}

export async function GET(req: Request) {
  try {
    // TODO: read venueId from session or querystring; fallback to first venue in session
    const url = new URL(req.url);
    const venueId = url.searchParams.get("venueId");

    const todayStart = startOfDay();
    const todayEnd = endOfDay();
    const weekStart = startOfWeek();

    // Revenue = sum(qty * sell price)
    const todaySalesRes = await db
      .select({ value: sql<number>`COALESCE(SUM(ds.quantity * mi.price), 0)` })
      .from(daily_sales as any) // alias if needed
      .leftJoin(menu_items as any, eq((daily_sales as any).menu_item_id, (menu_items as any).id))
      .where(and(
        gte((daily_sales as any).date, todayStart),
        lte((daily_sales as any).date, todayEnd),
        venueId ? eq((daily_sales as any).venue_id, venueId) : sql`TRUE`
      ));

    const wtdSalesRes = await db
      .select({ value: sql<number>`COALESCE(SUM(ds.quantity * mi.price), 0)` })
      .from(daily_sales as any)
      .leftJoin(menu_items as any, eq((daily_sales as any).menu_item_id, (menu_items as any).id))
      .where(and(
        gte((daily_sales as any).date, weekStart),
        lte((daily_sales as any).date, todayEnd),
        venueId ? eq((daily_sales as any).venue_id, venueId) : sql`TRUE`
      ));

    const salesToday = Number(todaySalesRes?.[0]?.value ?? 0);
    const salesWTD = Number(wtdSalesRes?.[0]?.value ?? 0);

    // Theoretical COGS% (if a calculated cost column exists on menu_items -> use it; else null)
    let cogsPct: number | null = null;
    try {
      const cogsRes = await db
        .select({ cost: sql<number>`COALESCE(SUM(ds.quantity * mi.theoretical_cost), 0)` })
        .from(daily_sales as any)
        .leftJoin(menu_items as any, eq((daily_sales as any).menu_item_id, (menu_items as any).id))
        .where(and(
          gte((daily_sales as any).date, weekStart),
          lte((daily_sales as any).date, todayEnd),
          venueId ? eq((daily_sales as any).venue_id, venueId) : sql`TRUE`
        ));
      const cost = Number(cogsRes?.[0]?.cost ?? 0);
      cogsPct = salesWTD > 0 ? +(100 * cost / salesWTD).toFixed(1) : null;
    } catch (_) {
      cogsPct = null;
    }

    // Scheduled labour % (sum scheduled wage cost this week / WTD sales)
    let labourPct: number | null = null;
    try {
      const labourRes = await db
        .select({
          value: sql<number>`COALESCE(SUM(EXTRACT(EPOCH FROM (sh.end_at - sh.start_at))/3600 * st.hourly_rate),0)`
        })
        .from(shifts as any)
        .leftJoin(staff as any, eq((shifts as any).staff_id, (staff as any).id))
        .where(and(
          gte((shifts as any).start_at, weekStart),
          lte((shifts as any).start_at, todayEnd),
          venueId ? eq((shifts as any).venue_id, venueId) : sql`TRUE`
        ));
      const labour = Number(labourRes?.[0]?.value ?? 0);
      labourPct = salesWTD > 0 ? +(100 * labour / salesWTD).toFixed(1) : null;
    } catch (_) {
      labourPct = null;
    }

    return NextResponse.json({
      salesToday, salesWTD, cogsPct, labourPct
    }, { status: 200 });
  } catch (e) {
    return NextResponse.json({ error: "kpi_error" }, { status: 500 });
  }
}

async function getKpis() {
  const res = await fetch(`${process.env.NEXT_PUBLIC_APP_URL ?? ""}/api/metrics/kpis`, { cache: "no-store" })
    .catch(() => null);
  if (!res || !res.ok) return { salesToday: 0, salesWTD: 0, cogsPct: null, labourPct: null };
  return res.json();
}

async function getSuggestions() {
  const res = await fetch(`${process.env.NEXT_PUBLIC_APP_URL ?? ""}/api/automation/suggestions`, { cache: "no-store" })

}
