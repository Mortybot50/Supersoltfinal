export const count_sessions = pgTable("count_sessions", {
  id: uuid("id").defaultRandom().primaryKey(),
  org_id: uuid("org_id").notNull(),
  venue_id: uuid("venue_id").notNull(),
  name: varchar("name", { length: 128 }).notNull(), // "Weekly Count – 2025-10-24"
  status: varchar("status", { length: 16 }).notNull().default("DRAFT"), // DRAFT|SUBMITTED|APPROVED|POSTED
  start_at: timestamp("start_at", { mode: "date" }).notNull(),
  end_at: timestamp("end_at", { mode: "date" }).notNull(),
  created_by: uuid("created_by").notNull(),
  submitted_at: timestamp("submitted_at", { mode: "date" }),
  approved_at: timestamp("approved_at", { mode: "date" }),
  posted_at: timestamp("posted_at", { mode: "date" })
});
createIndex("idx_counts_org_venue_status").on(count_sessions, t => [t.org_id, t.venue_id, t.status, t.start_at]);

export const count_lines = pgTable("count_lines", {
  id: uuid("id").defaultRandom().primaryKey(),
  session_id: uuid("session_id").notNull(),
  ingredient_id: uuid("ingredient_id").notNull(),
  // snapshot info to make the count self-contained:
  ingredient_name: varchar("ingredient_name", { length: 256 }).notNull(),
  base_uom: varchar("base_uom", { length: 8 }).notNull(),   // "g"|"ml"|"each"
  on_hand_before_base: integer("on_hand_before_base").notNull().default(0), // snapshot when session created
  theoretical_used_base: integer("theoretical_used_base").notNull().default(0), // computed usage for window
  counted_base: integer("counted_base").notNull().default(0), // user input
  variance_base: integer("variance_base").notNull().default(0), // counted - (on_hand_before - theoretical_used + receipts - other moves)
  notes: varchar("notes", { length: 512 })
});
createIndex("idx_count_lines_session").on(count_lines, t => [t.session_id, t.ingredient_id]);

export const waste_events = pgTable("waste_events", {
  id: uuid("id").defaultRandom().primaryKey(),
  org_id: uuid("org_id").notNull(),
  venue_id: uuid("venue_id").notNull(),
  ingredient_id: uuid("ingredient_id").notNull(),
  reason: varchar("reason", { length: 32 }).notNull(), // "prep_loss"|"expiry"|"spillage"|"overproduction"|"other"
  qty_base: integer("qty_base").notNull(),             // base units
  occurred_at: timestamp("occurred_at", { mode: "date" }).defaultNow().notNull(),
  notes: varchar("notes", { length: 256 })
});
createIndex("idx_waste_org_venue_ing_time").on(waste_events, t => [t.org_id, t.venue_id, t.ingredient_id, t.occurred_at]);

2) Count session APIs

Create routes:
	•	POST /api/counts/sessions
Body: { name?, start, end }
	•	Creates DRAFT session, snapshots on_hand_before_base at start, computes theoretical_used_base for each ingredient that is in use (sold or on hand), inserts count_lines. Returns { id }.
	•	GET /api/counts/sessions
Query: status?, from?, to? → list sessions.
	•	GET /api/counts/sessions/[id]
Returns header + lines.
	•	PATCH /api/counts/sessions/[id]/lines
Body: { lines: [{ lineId, counted_base, notes? }] }
	•	Updates counts client submitted; recomputes variance_base server-side.
	•	POST /api/counts/sessions/[id]/submit → status: SUBMITTED, submitted_at = now()
	•	POST /api/counts/sessions/[id]/approve → status: APPROVED, approved_at = now()
	•	POST /api/counts/sessions/[id]/post
	•	For each line, if variance_base !== 0, write stock_movements with type:"ADJUSTMENT" and qty_base = variance_base, unit_cost_cents = current_avg (if you track), ref_type:"COUNT", ref_id: session_id.
	•	Update on-hand and average cost accordingly (variance can be positive or negative).
	•	Set status: POSTED, posted_at = now().

All endpoints requireOrg({ role: "manager" }). Validate venue, time window sanity, and non-negative counts.

3) Waste logging API
	•	POST /api/inventory/waste
Body: { ingredientId, qty_base, reason, occurredAt?, notes? }
	•	Writes waste_events row.
	•	Also writes a stock_movements row: { type:"WASTE", qty_base: -Math.abs(qty_base), unit_cost_cents: current_avg, ref_type:"WASTE", ref_id:waste_id }.
	•	Updates on-hand (decrease) and keeps cost valuation correct.
	•	GET /api/inventory/waste?from=&to=&ingredientId? list/filter.
