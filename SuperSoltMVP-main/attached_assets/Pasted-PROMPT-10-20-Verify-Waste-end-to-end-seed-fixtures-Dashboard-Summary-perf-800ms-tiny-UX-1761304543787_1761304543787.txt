File: app/api/dev/seed/waste-fixtures/route.ts

export const runtime = "nodejs";
import { NextResponse } from "next/server";
import { requireOrg } from "@/lib/authz";
import { db } from "@/db";
// adjust imports to your schema names
import { ingredients, supplier_items, stock_movements } from "@/db/schema";
import { eq, and } from "drizzle-orm";

export async function POST() {
  const { venueId } = await requireOrg({ role: "manager" });

  // helper to upsert ingredient by (venue_id, name)
  async function upsertIngredient(name: string, baseUnit: "kg"|"each") {
    const existing = await db.query.ingredients.findFirst({
      where: (t,{and,eq}) => and(eq(t.venue_id, venueId), eq(t.name, name))
    });
    if (existing) return existing;
    const [row] = await db.insert(ingredients).values({
      venue_id: venueId,
      name,
      // minimal fields your schema expects:
      purchase_unit: baseUnit, // "kg" or "each"
      is_active: true,
    }).returning();
    return row;
  }

  async function ensureSupplierItem(ingredientId: string, packSize: number, packUnit: "kg"|"each", packCostCents: number, preferred = true) {
    const existing = await db.query.supplier_items.findFirst({
      where: (t,{and,eq}) => and(eq(t.venue_id, venueId), eq(t.ingredient_id, ingredientId))
    });
    if (existing) return existing;
    const [row] = await db.insert(supplier_items).values({
      venue_id: venueId,
      ingredient_id: ingredientId,
      supplier_name: "Fixture Foods",
      supplier_sku: `FX-${ingredientId.slice(0,6)}`,
      pack_size: packSize,
      pack_unit: packUnit,
      pack_cost_cents: packCostCents,
      preferred
    }).returning();
    return row;
  }

  // 1) Two common ingredients with realistic costs
  const chx = await upsertIngredient("Chicken Breast", "kg");    // will log waste in kg/g
  const lett = await upsertIngredient("Cos Lettuce", "each");    // waste in each

  await ensureSupplierItem(chx.id, 5, "kg", 5000, true);   // $50 per 5kg => $10/kg
  await ensureSupplierItem(lett.id, 12, "each", 1800, true); // $18 per 12 heads => $1.50/each

  // 2) Small on-hand stock so order guide/waste math isn’t negative-only
  //    (if you already track on-hand, these are safe no-ops from an accounting POV)
  await db.insert(stock_movements).values([
    { venue_id: venueId, ingredient_id: chx.id,  qty_base_units: 3000, unit: "g",   movement_type: "seed",  ref_id: null }, // +3kg
    { venue_id: venueId, ingredient_id: lett.id, qty_base_units: 10,   unit: "each",movement_type: "seed",  ref_id: null }, // +10 each
  ]);

  return NextResponse.json({ ok:true, ingredientIds: { chicken: chx.id, lettuce: lett.id }}, { status: 201 });
}
