0) Schema: settings + business hours (safe, additive)

File: db/schema.ts (append)

// Per-organisation defaults
export const organisation_settings = pgTable("organisation_settings", {
  org_id: uuid("org_id").primaryKey(),
  timezone: varchar("timezone", { length: 64 }).notNull().default("Australia/Melbourne"),
  target_cogs_pct: integer("target_cogs_pct").notNull().default(35),     // whole %
  target_labour_pct: integer("target_labour_pct").notNull().default(22), // whole %
  week_starts_on: integer("week_starts_on").notNull().default(1)         // 1 = Monday
});

// Per-venue settings (override org where set)
export const venue_settings = pgTable("venue_settings", {
  venue_id: uuid("venue_id").primaryKey(),
  timezone: varchar("timezone", { length: 64 }),
  display_name: varchar("display_name", { length: 128 }),
  safety_stock_days: integer("safety_stock_days"),
  default_order_window_days: integer("default_order_window_days"),
});

// Opening hours per day-of-week (0=Sun..6=Sat)
export const business_hours = pgTable("business_hours", {
  id: uuid("id").defaultRandom().primaryKey(),
  venue_id: uuid("venue_id").notNull(),
  dow: integer("dow").notNull(),
  open_minute: integer("open_minute").notNull(),   // minutes from midnight
  close_minute: integer("close_minute").notNull(), // minutes from midnight
});
createIndex("idx_hours_venue_dow").on(business_hours, t => [t.venue_id, t.dow]);


Run: npm run db:push

Seed defaults on first access if rows missing (org/venue).

1) Top-bar Venue Switcher

Files:

components/top-bar/VenueSwitcher.tsx (client)

app/api/session/active-venue/route.ts (POST sets cookie; GET returns active)

Behaviour

On load, read venueId from secure HTTP-only cookie.

Switcher lists venues user has membership for (query via requireOrg() + membership join).

Selecting a venue → POST /api/session/active-venue to set cookie, then router.refresh().

Enforcement

In your existing requireOrg()/requireVenue() helpers, add:

If no venueId cookie, redirect to /admin/locations (or a “Choose a venue” screen).

All data queries must include venue_id = activeVenueId.
