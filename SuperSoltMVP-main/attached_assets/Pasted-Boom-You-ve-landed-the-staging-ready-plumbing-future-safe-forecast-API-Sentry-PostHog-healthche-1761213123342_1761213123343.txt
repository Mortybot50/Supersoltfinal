Create a common parser + preview/commit pattern:

/api/import/templates              -> returns downloadable CSV templates (by type)

Files

	•	lib/imports/upsert.ts – upsert helpers with (org_id, venue_id, external_id) keys; return created/updated counts.

Preview endpoint (pattern)

// app/api/import/ingredients/preview/route.ts
export const runtime = "nodejs";
import { NextResponse } from "next/server";
import { requireOrg } from "@/lib/authz";
import { parseCsv } from "@/lib/imports/csv";
import { IngrRow } from "@/lib/imports/schemas";
import { normaliseIngr } from "@/lib/imports/normalise";

export async function POST(req: Request) {
  const { orgId, venueId } = await requireOrg({ role: "manager" });
  const text = await req.text();
  const { rows, errors } = parseCsv(text);
  const parsed = []; const rowErrors: any[] = [];
  for (let i=0;i<rows.length;i++){
    const r = rows[i];
    const v = IngrRow.safeParse(r);
    if (!v.success) { rowErrors.push({ i, error: v.error.flatten() }); continue; }
    const norm = normaliseIngr(v.data); // unit conversions etc.
    parsed.push(norm);
  }
  return NextResponse.json({ orgId, venueId, parsed, errors: rowErrors });
}

Commit endpoint (pattern)

// app/api/import/ingredients/commit/route.ts
export const runtime = "nodejs";
import { NextResponse } from "next/server";
import { requireOrg } from "@/lib/authz";
import { upsertIngredientsAndSuppliers } from "@/lib/imports/upsert";

export async function POST(req: Request) {
  const { orgId, venueId } = await requireOrg({ role: "manager" });
  const body = await req.json(); // expects { parsed: [...] }
  const res = await upsertIngredientsAndSuppliers({ orgId, venueId, rows: body.parsed });
  return NextResponse.json(res);
}
